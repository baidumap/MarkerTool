160         /**
161          * 注册对象的事件监听器。引入baidu.lang.Event后，Class的子类实例才会获得该方法。
162          * @grammar obj.addEventListener(type, handler[, key])
163          * @param 	{string}   type         自定义事件的名称
164          * @param 	{Function} handler      自定义事件被触发时应该调用的回调函数
165          * @param 	{string}   [key]		为事件监听函数指定的名称，可在移除时使用。如果不提供，方法会默认为它生成一个全局唯一的key。
166          * @remark 	事件类型区分大小写。如果自定义事件名称不是以小写"on"开头，该方法会给它加上"on"再进行判断，即"click"和"onclick"会被认为是同一种事件。 
167          */
168         baidu.lang.Class.prototype.addEventListener = function (type, handler, key) {
169             if (!baidu.lang.isFunction(handler)) {
170                 return;
171             }
172             !this.__listeners && (this.__listeners = {});
173             var t = this.__listeners, id;
174             if (typeof key == "string" && key) {
175                 if (/[^\w\-]/.test(key)) {
176                     throw("nonstandard key:" + key);
177                 } else {
178                     handler.hashCode = key; 
179                     id = key;
180                 }
181             }
182             type.indexOf("on") != 0 && (type = "on" + type);
183             typeof t[type] != "object" && (t[type] = {});
184             id = id || baidu.lang.guid();
185             handler.hashCode = id;
186             t[type][id] = handler;
187         };

189         /**
190          * 移除对象的事件监听器。引入baidu.lang.Event后，Class的子类实例才会获得该方法。
191          * @grammar obj.removeEventListener(type, handler)
192          * @param {string}   type     事件类型
193          * @param {Function|string} handler  要移除的事件监听函数或者监听函数的key
194          * @remark 	如果第二个参数handler没有被绑定到对应的自定义事件中，什么也不做。
195          */
196         baidu.lang.Class.prototype.removeEventListener = function (type, handler) {
197             if (baidu.lang.isFunction(handler)) {
198                 handler = handler.hashCode;
199             } else if (!baidu.lang.isString(handler)) {
200                 return;
201             }
202             !this.__listeners && (this.__listeners = {});
203             type.indexOf("on") != 0 && (type = "on" + type);
204             var t = this.__listeners;
205             if (!t[type]) {
206                 return;
207             }
208             t[type][handler] && delete t[type][handler];
209         };

211         /**
212          * 派发自定义事件，使得绑定到自定义事件上面的函数都会被执行。引入baidu.lang.Event后，Class的子类实例才会获得该方法。
213          * @grammar obj.dispatchEvent(event, options)
214          * @param {baidu.lang.Event|String} event 	Event对象，或事件名称(1.1.1起支持)
215          * @param {Object} options 扩展参数,所含属性键值会扩展到Event对象上(1.2起支持)
216          * @remark 处理会调用通过addEventListenr绑定的自定义事件回调函数之外，还会调用直接绑定到对象上面的自定义事件。
217          * 例如：<br>
218          * myobj.onMyEvent = function(){}<br>
219          * myobj.addEventListener("onMyEvent", function(){});
220          */
221         baidu.lang.Class.prototype.dispatchEvent = function (event, options) {
222             if (baidu.lang.isString(event)) {
223                 event = new baidu.lang.Event(event);
224             }
225             !this.__listeners && (this.__listeners = {});
226             options = options || {};
227             for (var i in options) {
228                 event[i] = options[i];
229             }
230             var i, t = this.__listeners, p = event.type;
231             event.target = event.target || this;
232             event.currentTarget = this;
233             p.indexOf("on") != 0 && (p = "on" + p);
234             baidu.lang.isFunction(this[p]) && this[p].apply(this, arguments);
235             if (typeof t[p] == "object") {
236                 for (i in t[p]) {
237                     t[p][i].apply(this, arguments);
238                 }
239             }
240             return event.returnValue;
241         };

246          * @function
247          * @grammar baidu.lang.inherits(subClass, superClass[, className])
248          * @param {Function} subClass 子类构造器
249          * @param {Function} superClass 父类构造器
250          * @param {string} className 类名标识
251          * @remark 使subClass继承superClass的prototype，
252          * 因此subClass的实例能够使用superClass的prototype中定义的所有属性和方法。<br>
253          * 这个函数实际上是建立了subClass和superClass的原型链集成，并对subClass进行了constructor修正。<br>
254          * <strong>注意：如果要继承构造函数，需要在subClass里面call一下，具体见下面的demo例子</strong>
255          * @shortcut inherits
256          * @meta standard
257          * @see baidu.lang.Class
258          */
259         baidu.lang.inherits = function (subClass, superClass, className) {
260             var key, proto, 
261                 selfProps = subClass.prototype, 
262                 clazz = new Function();        
263             clazz.prototype = superClass.prototype;
264             proto = subClass.prototype = new clazz();
265             for (key in selfProps) {
266                 proto[key] = selfProps[key];
267             }
268             subClass.prototype.constructor = subClass;
269             subClass.superClass = superClass.prototype;
270 
271             if ("string" == typeof className) {
272                 proto._className = className;
273             }
274         };

320     /**
321      * 开启工具
322      * @return {Boolean} true表示开启成功，false表示开启失败
323      */
324     MarkerTool.prototype.open = function(){
325         if(!this._map){
326             return false;
327         }
328         if (this._isOpen == true){
329           return true;
330         }
331         if (BMapLib._toolInUse){
332           return false;
333         }
334         BMapLib._toolInUse = true; //当前鼠标状态正在使用中，如果存在多个鼠标工具，
335                                      //可以使用此变量限制地图上只能存在一种鼠标操作状态
336         this._isOpen = true;
337 
338         // 绑定mousemove 和 click 事件
339         if (!this._binded){
340           this._bind();
341           this._binded = true;
342         }
343         //初始化跟随Marker
344         if (!this._followMarker) {
345           this._followMarker = new BMap.Marker(this._map.getCenter(), {offset: new BMap.Size(-10, -10)}); //偏移-10像素,解决cursor问题
346           this._map.addOverlay(this._followMarker);
347           this._followMarker.setZIndex(1000); // 设置跟随Marker的z轴高度
348           this._followMarker.hide();
349         }        
350 
351         //初始化跟随Label
352         if (!this._followLabel){              
353           this._followLabel = new BMap.Label(this._opts.followText, {offset: new BMap.Size(20, 0)});              
354         }
355         
356         this._preCursor = this._map.getDefaultCursor(); //记录当前的鼠标cursor
357         this._map.setDefaultCursor("url(" + MarkerTool.CUR_IMG + "), default");//设置鼠标样式            
358         
359         return true;
360     };

408     /**
409      * 绑定地图的mousemove 和 click 事件
410      * @return 无返回值
411      */
412     MarkerTool.prototype._bind = function(){
413         var me = this;            
414         if (!me._isOpen){//判断工具是否打开
415             return;
416         }
417         
418         //绑定mousemove事件
419         me._mouseMoveHandler = function(evt){
420             var pt = evt.point;
421             me._followMarker.setIcon(me._opts.icon); //每次都检查最新的Icon
422             me._followMarker.setPosition(pt);
423             me._followMarker.setLabel(me._followLabel);
424             me._followMarker.show();            
425         }        
426         me._map.addEventListener("mousemove", me._mouseMoveHandler);
427         
428         //绑定click事件
429         me._clickHandler = function(evt){
430             var evtPix = evt.pixel;
431             var iconPix = new BMap.Pixel(evtPix.x - 10, evtPix.y - 10); //补偿_followMarker的-10像素问题,解决cursor问题
432             var pt = me._map.pixelToPoint(iconPix);
433             
434             var mkr = new BMap.Marker(pt, {icon: me._opts.icon}); 
435             me._map.addOverlay(mkr);
436             
437             /**
438              * 添加标注过程中，每次点击地图添加完标注时，派发事件的接口
439              * @name MarkerTool#onmarkend
440              * @event
441              * @param {Event Object} e 回调函数会返回event参数，包括以下返回值：
442              * <br />"<b>type</b> : {String} 事件类型
443              * <br />"<b>target</b>：{MarkerTool} 当前MarkerTool对象
444              * <br />"<b>marker</b>：{Marker} 当前添加的Marker标注             
445              *
446              * @example <b>参考示例：</b><br />
447              * mkrTool.addEventListener("markend", function(e) {  alert(e.type);  });
448              */            
449             var event = new baidu.lang.Event("onmarkend");
450             event.marker = mkr;
451             me.dispatchEvent(event);
452 
453             if(me._opts.autoClose){ //自动关闭工具
454                 me.close();
455             }
456         }
457         me._map.addEventListener("click", me._clickHandler);
458     };
